		$("td, th").mouseover(function(e) {
			if (this.offsetWidth < this.scrollWidth) {
				//console.log("此单元框内的内容已溢出!!!");

				var myTitle = $(this).text();
				var hint = "<div id='hint'>" + myTitle + "</div>";
				$("body").append(hint);

				$("#hint").css({
					"top": (e.pageY + y) + "px",
					"left": (e.pageX - x) + "px"
				}).show("fast");
			}

		}).mouseout(function() {
			$("#hint").remove();

		}).mousemove(function(e) {
			$("#hint").css({
				"top": (e.pageY + y) + "px",
				"left": (e.pageX - x) + "px"
			});
		});

	})
